lifedrain_prisoner_interaction = {
	interface_priority = 30
	category = interaction_category_religion
	desc = lifedrain_prisoner_interaction_desc
	is_shown = {
		scope:actor = {
			has_trait = blood_mage
		}
		scope:recipient = {
			is_imprisoned_by = scope:actor
		}
	}
	on_accept = {
		scope:recipient = {
			send_interface_toast = {
				left_icon = scope:actor
				custom_tooltip = lifedrain_prisoner_interaction.recipient.notification.tt
			}
		}
		scope:actor = {
			drain_lifeforce_effects = yes
		}
	}
	auto_accept = yes
}

# Gain quick / intelligent / genius traits
intellect_drain_prisoner_interaction = {
	interface_priority = 50
	category = interaction_category_religion
	desc = intellect_drain_prisoner_interaction_desc
	#################################
	# Show only if prisoner has trait
	is_shown = {
		scope:actor = {
			has_trait = blood_mage
		}
		scope:recipient = {
			AND = {
				is_imprisoned_by = scope:actor
				has_trait = intellect_good
			}
		}
	}
	#################################
	# Highlight
	is_highlighted = {
		scope:actor = {
			has_trait = blood_mage
		}
	}
	on_accept = {
		scope:recipient = {
			send_interface_toast = {
				left_icon = scope:actor
				custom_tooltip = intellect_drain_prisoner_interaction.recipient.notification.tt
			}
		}
		scope:actor = {
			# Carry out logic
			drain_trait_intellect_effects = yes
		}
	}
	auto_accept = yes
	cost = {
		piety = 1000
	}
}
