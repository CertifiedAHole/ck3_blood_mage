lifedrain_interaction = {
	interface_priority = 999

	category = interaction_category_prison

	desc = lifedrain_prison_interaction_desc

	is_shown = {
		scope:recipient = {
			is_imprisoned_by = scope:actor
		}
		scope:actor = {
			is_adult = yes
		}
	}

	is_valid_showing_failures_only = {
		scope:recipient = {
			custom_description = {
				text = "currently_being_tortured"
				NOT = { has_character_flag = is_being_tortured }
			}
		}
	}

	is_highlighted = {
		OR = {
			scope:actor = { has_trait = blood_mage }
			scope:actor = { has_relation_rival = scope:recipient }
			scope:recipient = {
				any_claim = {
					holder = scope:actor
				}
			}
		}
	}

	on_accept = {
		scope:recipient = {
			add_character_flag = {
				flag = is_being_tortured
				months = 6 # Players in MP can keep the event window open...
			}
		}

		blind_and_castrate_actor_effect = yes #Stress & dread

		show_as_tooltip = {
			scope:recipient = {
				add_trait = blind
				release_from_prison = yes
			}
		}

		torture_blind_castrate_opinion_effect = { VERB = blinded }

		scope:actor = {
			trigger_event = {
				id = prison.1020
				days = 1
			}
		}
	}

	auto_accept = yes

	ai_targets = {
		ai_recipients = prisoners
	}

	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 20

	}

	ai_frequency = 20
}
